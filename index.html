// ── HELPERS ─────────────────────────────────────────────────────────────────

function getScoreColor(s) {

  if (!s) return "#888";

  if (s >= 8) return "#196F3D";

  if (s >= 5) return "#E67E22";

  return "#C1272D";

}

function getScoreBg(s) {

  if (!s) return "#F5F5F5";

  if (s >= 8) return "#EAFAF1";

  if (s >= 5) return "#FEF9E7";

  return "#FDEDEC";

}

function calcWeightedScore(scores) {

  let answered = CRITERIA.filter(c => scores[c.id]);

  if (!answered.length) return null;

  let totalWeight = answered.reduce((a, c) => a + c.weight, 0);

  let weighted    = answered.reduce((a, c) => a + (scores[c.id] * c.weight), 0);

  return (weighted / totalWeight).toFixed(1);

}


// ── COMPONENTS ──────────────────────────────────────────────────────────────

function WeightBar({ weight, color }) {

  return (

    <div className="wbar-wrap">

      <div className="wbar" style={{ width: weight * 4, background: color, opacity: 0.85 }} />

      <span className="wbar-pct">{weight}%</span>

    </div>

  );

}


function ScoreBadge({ score }) {

  if (!score) return null;

  const c = getScoreColor(score);

  const b = getScoreBg(score);

  return (

    <span className="score-badge" style={{ color: c, background: b, borderColor: c }}>

      {score}/10

    </span>

  );

}

function EvaluadorTab() {

  const [scores, setScores] = useState(() => {

    try { return JSON.parse(localStorage.getItem("ux_scores") || "{}"); }

    catch { return {}; }

  });


  useEffect(() => {

    localStorage.setItem("ux_scores", JSON.stringify(scores));

  }, [scores]);


  const answered  = Object.keys(scores).length;

  const adjScore  = calcWeightedScore(scores);

  const sc        = adjScore ? getScoreColor(parseFloat(adjScore)) : "#888";

  const sb        = adjScore ? getScoreBg(parseFloat(adjScore)) : "#F5F5F5";


  function setScore(id, n) {

    setScores(prev => ({ ...prev, [id]: n }));

  }

  function reset() {

    if (confirm("¿Reiniciar toda la evaluación?")) {

      setScores({});

      localStorage.removeItem("ux_scores");

    }

  }


  return (

    <div className="eval-header">

      <div className="eval-top">

        <div>

          <div className="eval-title">Evaluador Interactivo</div>

          <div className="eval-sub">Califica de 1 a 10 · Tu progreso se guarda automáticamente en este navegador</div>

        </div>

        {adjScore && (

          <div className="score-display" style={{ borderColor: sc, background: sb }}>

          <div className="breakdown-title">Desglose por criterio</div>
